```python
"""CLI entry point for AI Readiness Checker.

Run: python -m checker <url> [--timeout SECONDS] [--verbose]
"""

import sys


def main(argv: list[str] | None = None) -> int:
    """Parse CLI arguments, run the analysis pipeline, and display results.

    Args:
        argv: Argument list (defaults to sys.argv[1:] when None).

    Returns:
        Exit code: 0 on success.
    """
    import argparse
    from .orchestrator import run_pipeline
    from .cli_renderer import display_score_card

    parser = argparse.ArgumentParser(
        prog="python -m checker",
        description=(
            "AI Readiness Checker — score any website's AI search engine visibility"
        ),
        epilog="Example: python -m checker https://example.com",
    )
    parser.add_argument(
        "url",
        help="URL to analyze (e.g., https://example.com)",
    )
    parser.add_argument(
        "--timeout", "-t",
        type=float,
        default=10.0,
        help="HTTP request timeout in seconds (default: 10)",
    )
    parser.add_argument(
        "--verbose", "-v",
        action="store_true",
        help="Show detailed per-stage analysis progress",
    )
    args = parser.parse_args(argv)

    # Run the full analysis pipeline
    result = run_pipeline(args.url, timeout=args.timeout, verbose=args.verbose)

    # Display the formatted score card
    display_score_card(result)

    return 0


if __name__ == "__main__":
    sys.exit(main())
```

Key implementation details:
- `main()` accepts optional `argv: list[str] | None` parameter for testability — when None, argparse reads sys.argv[1:] by default. When list is provided, argparse parses that instead. This allows tests to call `main(["https://example.com"])` without patching sys.argv.
- `prog="python -m checker"` — explicit program name for help output
- `sys.exit(main())` pattern — enables shell scripts to check `$?` exit code
- All imports inside `main()` — follows the pattern of `importlib.metadata` deferred import for fast startup and testability (imports only resolved when main() is called)

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Terminal argv -> argparse | User-supplied URL string and flags from command line |
| argparse -> orchestrator.run_pipeline() | Parsed URL and timeout/verbose values passed as typed arguments |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-06-09 | Tampering | argparse URL argument | mitigate | URL is parsed as string by argparse — no shell interpolation or command injection possible since argparse positional args are plain strings; downstream SSRF prevention in crawler.py is_ssrf_safe() blocks internal addresses |
| T-06-10 | Tampering | --timeout argument type coercion | mitigate | argparse `type=float` validates numeric input; invalid values (letters, negative numbers) cause argparse to print error and exit with code 2; no code path allows crafting timeout to bypass crawler limit |
| T-06-11 | Information Disclosure | --help output | accept | --help only shows static description text and argument names; no dynamic content, no system paths, no environment variables exposed |
| T-06-12 | Denial of Service | sys.exit() return code | accept | exit code 0 on success, 2 on argparse error (standard); no information leakage through exit codes |
</threat_model>
