```python
def render_score_hero(result: dict) -> None:
    """Render the overall score metric and grade badge."""
    report = result["report"]
    grade = report.grade
    
    # Column layout: score on left, grade badge on right
    col_score, col_grade = st.columns([2, 1])
    
    with col_score:
        st.metric(
            label="Overall Score",
            value=f"{report.overall_score}/100",
            border=True,
        )
    
    with col_grade:
        color_hex = GRADE_HEX.get(grade, "#6c757d")
        st.markdown(f"""
        <div style="
            display: inline-block;
            background-color: {color_hex};
            color: white;
            font-size: 48px;
            font-weight: 700;
            line-height: 1.1;
            padding: 8px 24px;
            border-radius: 8px;
            text-align: center;
            min-width: 80px;
        ">{grade}</div>
        """, unsafe_allow_html=True)
```

The grade badge uses inline HTML with `unsafe_allow_html=True` because Streamlit's native `st.metric` does not support custom colored backgrounds. The grade value (`A`/`B`/`C`/`D`/`F`) comes from `ScoreReport.grade`, which is always one of these 5 literal strings from `scorer.py` constants -- never user-controlled content. The color hex is from the hardcoded `GRADE_HEX` dict. No XSS vector exists.

```python
def render_module_expanders(result: dict) -> None:
    """Render per-module expandable sections with score bars and detail content."""
    report = result["report"]
    st.subheader("Module Breakdown")
    
    for module_key in MODULE_ORDER:
        data = report.module_breakdown.get(module_key)
        if data is None:
            continue
        score = data["score"]
        display_name = MODULE_DISPLAY_NAMES[module_key]
        
        with st.expander(f"{display_name} (score: {score:.2f})", expanded=False):
            st.progress(score)
            _render_module_detail(module_key, result)


def _render_module_detail(module_key: str, result: dict) -> None:
    """Render module-specific detail content inside an expander."""
    if module_key == "robots":
        _render_robots_detail(result)
    elif module_key == "llms_txt":
        _render_llms_detail(result)
    elif module_key == "schema":
        _render_schema_detail(result)
    elif module_key == "content":
        _render_content_detail(result)
```

All expanders are `expanded=False` by default (collapsed), matching the UI-SPEC wireframe which shows all expanders with the collapsed "▶" symbol.

```python
def _render_robots_detail(result: dict) -> None:
    """Render robots.txt bot status table inside the Robots expander."""
    robots = result.get("robots_result")
    if robots is None or not robots.bots:
        st.write("No bot data available.")
        return
    
    if not robots.exists:
        st.write("robots.txt not found.")
        return
    
    # Build a markdown table: Bot Name | Status | Rule
    lines = ["| Bot | Status | Rule |", "|-----|--------|------|"]
    for bot in robots.bots:
        status_color = STATUS_HEX.get(bot.status, "#6c757d")
        status_badge = (
            f'<span style="color:{status_color};font-weight:600;">'
            f'{bot.status.replace("_", " ").title()}</span>'
        )
        rule = bot.rule_line if bot.rule_line else "—"
        lines.append(f"| {bot.bot_name} | {status_badge} | {rule} |")
    
    st.markdown("\n".join(lines), unsafe_allow_html=True)


def _render_llms_detail(result: dict) -> None:
    """Render llms.txt details inside the llms.txt expander."""
    llms = result.get("llms_result")
    if llms is None:
        st.write("No data available.")
        return
    
    if llms.found:
        st.write("**Status:** Found")
        if llms.valid is not None:
            valid_text = "Valid" if llms.valid else "Invalid"
            valid_color = "#2ecc71" if llms.valid else "#e74c3c"
            st.markdown(f'**Format:** <span style="color:{valid_color};">{valid_text}</span>', unsafe_allow_html=True)
        if llms.validation_errors:
            st.write("**Validation Errors:**")
            for err in llms.validation_errors:
                st.markdown(f"- {err}")
        if llms.content_preview:
            st.write("**Content Preview:**")
            st.code(llms.content_preview, language="markdown")
        if llms.raw_text:
            st.write(f"**Full Content** ({len(llms.raw_text)} chars):")
            st.code(llms.raw_text, language="markdown")
    else:
        st.write("**Status:** Not Found")
        if llms.fetch_error:
            st.write(f"**Error:** {llms.fetch_error}")


def _render_schema_detail(result: dict) -> None:
    """Render schema extraction details inside the Schema expander."""
    schema = result.get("schema_analysis")
    if schema is None:
        st.write("No data available.")
        return
    
    # Detected types
    if schema.detected_types:
        st.write("**Detected Schema Types:**")
        for stype in sorted(schema.detected_types):
            details = schema.type_details.get(stype, {})
            count = details.get("count", 0)
            formats = details.get("formats", [])
            fmt_str = ", ".join(formats) if formats else "—"
            st.markdown(f"- **{stype}** ({count} instance(s), formats: {fmt_str})")
    else:
        st.write("No schema types detected.")
    
    # Raw format breakdown
    if schema.raw:
        st.write("**Raw Format Breakdown:**")
        for fmt_name, items in schema.raw.items():
            st.write(f"- {fmt_name}: {len(items)} block(s)")


def _render_content_detail(result: dict) -> None:
    """Render content analysis details inside the Content expander."""
    content = result.get("content_analysis")
    if content is None:
        st.write("No data available.")
        return
    
    # Readability scores
    st.write("**Readability**")
    st.write(f"- Flesch Reading Ease: {content.flesch_raw:.1f}")
    st.write(f"- Gunning Fog Index: {content.fog_raw:.1f}")
    
    # Text ratio
    st.write(f"**Content-to-HTML Ratio:** {content.raw_text_ratio:.1%}")
    
    # Entity breakdown
    if content.entities:
        st.write("**Named Entities:**")
        for entity_type in ["ORG", "PRODUCT", "GPE", "PERSON"]:
            entities_list = content.entities.get(entity_type, [])
            if entities_list:
                preview = ", ".join(entities_list[:5])
                suffix = f" and {len(entities_list) - 5} more" if len(entities_list) > 5 else ""
                st.write(f"- {entity_type}: {preview}{suffix}")
            else:
                st.write(f"- {entity_type}: none")
    
    # Heading structure
    ha = content.heading_analysis
    if ha:
        st.write("**Heading Structure:**")
        st.write(f"- H1 count: {ha.get('h1_count', 0)}")
        st.write(f"- H1 unique: {'Yes' if ha.get('h1_unique', False) else 'No'}")
        st.write(f"- H2 count: {ha.get('h2_count', 0)}")
        st.write(f"- H3 count: {ha.get('h3_count', 0)}")
        violations = ha.get('hierarchy_violations', [])
        if violations:
            st.write(f"- Hierarchy violations: {len(violations)}")
    
    # Q&A density
    qa = content.qa_analysis
    if qa:
        st.write("**Q&A Density:**")
        st.write(f"- Questions: {qa.get('question_count', 0)}")
        st.write(f"- Answers: {qa.get('answer_count', 0)}")
        total = qa.get('total_sentences', 1)
        st.write(f"- Total sentences: {total}")
```

```python
def render_recommendations(result: dict) -> None:
```

Recommendation `message` values come from `scorer.py` string literals, not user input. No XSS risk from rendering these in HTML. Priority values are always "HIGH", "MEDIUM", or "LOW" -- also string literals.

```python
def render_errors(result: dict) -> None:
    """Render pipeline errors. Show verbatim error strings (per D-04)."""
    errors = result.get("errors", [])
    if not errors:
        return  # Empty state: render nothing
    
    st.error("Analysis failed for some stages. See details below.")
    for error in errors:
        st.markdown(f"- {error}")
```

The error banner text "Analysis failed for some stages. See details below." comes from `.planning/phases/07-streamlit-dashboard/07-UI-SPEC.md` line 132. Error messages are rendered with plain `st.markdown(f"- {error}")` -- NO `unsafe_allow_html=True` on error rendering. This prevents XSS if an error message happens to contain HTML from a crawled page. The error strings are displayed verbatim as text (per D-04).

**Main app flow (after all render functions):**
```python
# --- Main layout ---
st.title("AI Readiness Checker")

url = st.text_input(
    "Enter a URL to analyze",
    placeholder="https://example.com",
    key="url_input",
)

col_input, col_spacer = st.columns([1, 4])
with col_input:
    analyze_clicked = st.button("Analyze", type="primary")

if analyze_clicked and url:
    st.session_state.current_url = url
    st.session_state.analysis_done = True
elif analyze_clicked and not url:
    st.warning("Please enter a URL.")

# --- Results area (only if analysis has been run) ---
if st.session_state.get("analysis_done"):
    result = analyze_url(st.session_state.current_url)
    
    render_score_hero(result)
    st.markdown("<br>", unsafe_allow_html=True)  # lg spacing (24px)
    render_module_expanders(result)
    st.markdown("<br>", unsafe_allow_html=True)  # xl spacing (32px)
    render_recommendations(result)
    st.markdown("<br>", unsafe_allow_html=True)
    render_errors(result)
```

Key architectural decisions in this main flow:

1. **`st.session_state` stores only metadata** (current_url string, analysis_done boolean) -- never the pipeline result dict. Follows D-02: "No st.session_state needed for the result."
2. **On every rerun** (expander click, scroll, etc.), the conditional `if st.session_state.get("analysis_done")` re-calls `analyze_url(st.session_state.current_url)`. `@st.cache_data` returns instantly from cache (no pipeline re-run). This is the core of DASH-06.
3. **On first run**, `st.button("Analyze")` sets `st.session_state.current_url = url`. The script continues, `st.session_state.analysis_done` is now True, the results conditional triggers, `analyze_url(url)` has a cache miss, the pipeline runs, the spinner shows.
4. **Empty state**: Before first analysis, `st.session_state.get("analysis_done")` returns None (falsy), so no results area renders. Only the title, URL input, and Analyze button are visible. Matches UI-SPEC Pattern 4 (no placeholder content).
5. **Button label**: "Analyze" (per UI-SPEC line 127, Primary CTA).
6. **No-URL guard**: `elif analyze_clicked and not url:` shows `st.warning("Please enter a URL.")` per UI-SPEC line 133.
7. **Spacing**: `<br>` tags between results sections approximate the spacing scale from UI-SPEC (lg=24px, xl=32px). Streamlit doesn't have a native spacing API; `st.markdown("<br>")` is the standard workaround for vertical spacing between sections.

**No `if __name__ == "__main__"` guard:** Streamlit scripts execute top-to-bottom when run with `streamlit run app.py`. The `__name__ == "__main__"` guard is not needed and would prevent Streamlit from executing the script correctly.

**No `st.sidebar` usage:** The sidebar is collapsed by default (`initial_sidebar_state="collapsed"` in `st.set_page_config`). The dashboard has no navigation or filtering that would benefit from a sidebar.

**Imports are at module level:** All imports are at the top of the file. Streamlit re-executes the entire script on each rerun, but Python's import cache makes repeated imports essentially free. The `@st.cache_data` decorator is applied at module level as required by Streamlit.

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| User Browser -> Streamlit Server | URL input crosses from browser to server via Streamlit websocket. This is the primary untrusted input surface. |
| Streamlit Server -> External Web | Pipeline fetches content from user-supplied URLs. Trust boundary between the local process and the internet. |
| st.markdown(unsafe_allow_html=True) -> Browser DOM | Inline HTML from Python string literals injected into the browser DOM. Trust boundary between hardcoded strings and potential user-controlled content sneaking in. |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-07-02 | Spoofing | URL text input | mitigate | `crawler.fetch_url()` already validates URLs and blocks SSRF (localhost, private networks). The dashboard passes the raw string to `run_pipeline()` which delegates validation to the crawler. No additional validation needed at the dashboard layer. |
| T-07-03 | Information Disclosure | `st.markdown(unsafe_allow_html=True)` with grade badge | mitigate | Grade badge HTML uses `GRADE_HEX[grade]` where `grade` is always one of `A`/`B`/`C`/`D`/`F` from `scorer.py` constants (string literals). The grade value never contains user input. The color hexes are hardcoded module-level constants. No injection vector. |
| T-07-04 | Information Disclosure | `st.markdown(unsafe_allow_html=True)` with recommendations | mitigate | Recommendation `message` and `priority` fields come from `scorer.py` string literals, not from crawled page content. Priority values are always `HIGH`/`MEDIUM`/`LOW` (validated enum). No injection vector. |
| T-07-05 | Information Disclosure | `st.markdown(unsafe_allow_html=True)` with bot status table | mitigate | Bot `status` values come from `robots_txt.py` which returns only `"allowed"`, `"blocked"`, or `"not_mentioned"`. Bot `name` is from a hardcoded list of 7 bot names. Rule line is extracted from robots.txt content (comes from external site) but is rendered within a `<span>` tag with no executable attributes. |
| T-07-06 | Information Disclosure | Error messages rendered via `st.markdown(f"- {error}")` | mitigate | Error strings come from pipeline exceptions and could theoretically contain HTML from a malicious fetched page. However, error rendering uses `st.markdown(f"- {error}")` WITHOUT `unsafe_allow_html=True`. Markdown escaping is applied automatically by Streamlit. No HTML injection possible through error display. |
| T-07-07 | Denial of Service | Repeated Analyze button clicks with different URLs | accept | A user could submit many unique URLs to exhaust memory via `st.cache_data` growth. This is a local development tool with no multi-tenant exposure. Cache is bounded by Streamlit's default TTL and memory limits. A malicious user would be attacking their own machine. Accept as low risk. |
| T-07-08 | Elevation of Privilege | `st.cache_data` pickled result deserialization | accept | `st.cache_data` uses `pickle` internally. The cached data is the 8-key dict from `run_pipeline()`, which contains only project-defined dataclasses and standard Python types (str, float, int, list, dict, datetime). No arbitrary objects enter the cache. Risk is theoretical only -- an attacker would need filesystem access to tamper with the pickle cache, at which point they already own the machine. |
